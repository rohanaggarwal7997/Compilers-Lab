PROG -> LIBRARIES FUNCTIONS INT MAIN LB_ROUND ARGS_LIST_DECLR RB_ROUND LB_CURLY STATEMENTS RB_CURLY
LIBRARIES -> LIBRARY LIBRARIES| epsilon
FUNCTIONS -> FUNCTION_DECLR FUNCTIONS | epsilon

DATA_TYPE -> INT | VOID | CHR | BOOL | STR | FLOAT | INT TIMES | CHR TIMES | FLOAT TIMES | STR TIMES

ARGS_LIST_DECLR -> DATA_TYPE ID COMMA ARGS_LIST_DECLR | EPSILON

ARGS_LIST -> ID COMMA ARGSLIST | EPSILON

FUNCTION_DECLR -> DATA_TYPE ID LB_ROUND ARGS_LIST_DECLR RB_ROUND LB_CURLY STATEMENTS RB_CURLY

STATEMENTS -> STATEMENT STATEMENTS | epsilon

STATEMENT -> IFSTATEMENT| LOOP | DECLARATION SEMI| ASSIGNMENT_NESTED SEMI| EXPRESSION SEMI

EXPRESSION -> TERM EXPRESSION1
EXPRESSION1 -> PLUS TERM EXPRESSION1 | MINUS TERM EXPRESSION1 | EPSILON

TERM -> FACTOR TERM1
TERM1 -> TIMES FACTOR TERM1 | DIVIDE FACTOR TERM1 | EPSILON

FACTOR -> NUM_OR_ID | LB_ROUND EXPRESSION RB_ROUND | FUNCTION_CALL | CONSTRUCTORS | MEMBER_FUNCTIONS




IFSTATEMENT -> IF LB_ROUND EXPRESSION RB_ROUND LB_CURLY STATEMENTS RB_CURLY |
				IF LB_ROUND EXPRESSION RB_ROUND LB_CURLY STATEMENTS RB_CURLY ELSE LB_CURLY STATEMENTS RB_CURLY |
				IF LB_ROUND EXPRESSION RB_ROUND LB_CURLY STATEMENTS RB_CURLY ELSE IFSTATEMENT

LOOP -> WHILELOOP | FORLOOP

WHILELOOP -> WHILE LB_ROUND EXPRESSION RB_ROUND LB_CURLY STATEMENTS RB_CURLY
FORLOOP -> FOR LB_ROUND DECLARATION SEMI EXPRESSION SEMI EXPRESSION RB_ROUND LB_CURLY STATEMENTS RB_CURLY

DECLARATION -> DECLR_1 | DECLR_2 | DECLR_3 | DECLR_4 | DECLR_5

DECLR_1 -> INT ID EQUAL EXPRESSION | BOOL ID EQUAL EXPRESSION | FLOAT ID EQUAL EXPRESSION | INT ID | BOOL ID | FLOAT ID
			|  INT ID EQUAL ASSIGNMENT_NESTED | BOOL ID EQUAL ASSIGNMENT_NESTED | FLOAT ID EQUAL ASSIGNMENT_NESTED

DECLR_2 -> STR ID EQUAL STRING | STR ID | STR ID EQUAL ASSIGNMENT_NESTED
DECLR_3 -> CHR ID EQUAL CHAR | CHR ID | CHR ID EQUAL ASSIGNMENT_NESTED
DECLR_4 -> INT TIMES ID | FLOAT TIMES ID | CHAR TIMES ID | STR TIMES ID
DECLR_5 -> INT ID LB_SQUARE NUM_OR_ID RB_SQUARE | FLOAT ID LB_SQUARE NUM_OR_ID RB_SQUARE | CHAR ID LB_SQUARE NUM_OR_ID RB_SQUARE


ASSIGNMENT_NESTED -> ID EQUAL ASSIGNMENT_NESTED | ASSIGNMENT | ASSIGNMENT_ARRAY | ASSIGNMENT_POINTER

ASSIGNMENT -> ID EQUAL EXPRESSION

ASSIGNMENT_ARRAY -> ID LB_SQUARE NUM_OR_ID RB_SQUARE EQUAL EXPRESSION

ASSIGNMENT_POINTER -> ID EQUAL AMP ID | TIMES ID EQUAL NUM_OR_ID | TIMES ID EQUAL TIMES ID


FUNCTION_CALL -> ID LB_ROUND ARGS_LIST RB_ROUND

ARRAY -> LB_SQUARE ARRAY_LIST RB_SQUARE
ARRAY_LIST -> EXPRESSION COMMA ARRAY_LIST | EPSILON
